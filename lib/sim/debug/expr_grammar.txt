expr     = p7
p7       = p6 (&|^ p7)?
p6       = p5 ([equality] expr)?
p5       = p4 ([inequality] expr)?
p4       = p3 (<<|>> expr)?
p3       = p2 ([+-] expr)?
p2       = p1 ([*/%] expr)?
p1       = [*+-~!&]? p0
p0       = value | IDENT ( . | ->) IDENT
value    = register | constant | # IDENT | IDENT | ( expr )
constant = [0-9]+ | 0[xX][0-9a-fA-F]+ | 0[bB][01]+
register = $ IDENT
